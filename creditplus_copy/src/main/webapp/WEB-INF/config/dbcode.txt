DROP TABLE EVAL_CON PURGE;
DROP TABLE ACCOUNT_ITEM PURGE;
DROP TABLE COMPANY PURGE;

CREATE TABLE COMPANY (
    EMPNO VARCHAR2(10) CONSTRAINT COMPANY_PK PRIMARY KEY,
    EMPNAME VARCHAR2(100),
    EMPDIV VARCHAR2(1),
    MODELDIV VARCHAR2(1),
    GRADE VARCHAR2(1)
);

CREATE TABLE ACCOUNT_ITEM (
    ACCID VARCHAR2(10) CONSTRAINT ACCOUNT_ITEM_PK PRIMARY KEY,
    ACCNAME VARCHAR2(45)
);

CREATE TABLE EVAL_CON (
    EVAID VARCHAR2(10),
    EMPNO VARCHAR2(10),
    ACCID VARCHAR2(10),
    PRICE VARCHAR2(18),
    EVADATE TIMESTAMP(9),
    EVASTATE VARCHAR2(1),
    CONSTRAINT EVAL_CON_PK PRIMARY KEY (EVAID, ACCID, EMPNO),
    CONSTRAINT ACCID_FK FOREIGN KEY(EMPNO) REFERENCES COMPANY(EMPNO),
    CONSTRAINT EMPNO_FK FOREIGN KEY(ACCID) REFERENCES ACCOUNT_ITEM(ACCID)
);
